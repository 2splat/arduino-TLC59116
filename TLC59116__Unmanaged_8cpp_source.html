160 
161  // Iout = ((1.26v * ((1 + HC) * (1 + D/64) / 4)) / Rext) * 15 * 3^(CM-1)
162  // (as per specsheet "SLDS157D – FEBRUARY 2008 – REVISED JULY 2011")
163  // where D is CC bit-reversed
164  // simplify for HC & CM and D=0|127 to get ranges below
165 
166  byte actual = 255; // anything is better.

168  HC=0; CM=0; D=0;

171  }
172  else if (ma >= 120 || ma > 28202.3/Rext) {

176  }

181  // printf("segment 1, actual %dma HC %d, CM %d, D %03o\n",actual, HC, CM, D);
182  }
183  else if (ma <= 14101.2/Rext) {
184  HC=0; CM=1; D = (2.560*ma*Rext)/189-64;
